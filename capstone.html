Example Endpoint

Lesson completion route: updates progress and creates an engagement entry. 

// POST /api/lessons/:lessonId/complete
router.post('/api/lessons/:lessonId/complete', verifyToken, async (req, res) => {
  const { lessonId } = req.params;
  const userId = req.user.id;

  await Progress.updateOne(
    { user: userId, lesson: lessonId },
    { $set: { completed: true, completedAt: new Date() } },
    { upsert: true }
  );

  await CRMEntry.create({
    user: userId,
    lesson: lessonId,
    engagementScore: 10,
    note: 'Completed lesson',
    timestamp: new Date()
  });

  res.status(200).json({ message: 'Lesson marked complete' });
});
